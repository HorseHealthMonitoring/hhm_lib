            ESOS_TASK_WAIT_ON_SEND_STRING(psz_buttonDoublePressPeriods);
            ESOS_TASK_WAIT_ON_SEND_STRING(psz_CRNL);

	    ESOS_TASK_SIGNAL_AVAILABLE_OUT_COMM();

	        ESOS_TASK_WAIT_ON_AVAILABLE_IN_COMM();

	    ESOS_TASK_WAIT_ON_SEND_STRING(psz_CRNL);

            ESOS_TASK_SPAWN_AND_WAIT(th_child, buttonDoublePressPeriodsScreenMain_child, u8_mainScreenChoice-'0');
	    }
